def NaiveInteract(types,Depth):

    classifier1 = DecisionTreeClassifier(criterion=types,max_depth=Depth)
    classifier1.fit(Xtrain,ytrain)
    
    x_min1, x_max1 = Xtrain[:,0].min() - 0.1 , Xtrain[:,0].max() + 0.1
    y_min1, y_max1 = Xtrain[:,1].min() - 0.1 , Xtrain[:,1].max() + 0.1
    xx1,yy1 = np.meshgrid(np.arange(x_min1,x_max1,0.01),np.arange(y_min1,y_max1,0.01))
    z1 = classifier1.predict(np.c_[xx1.ravel(),yy1.ravel()])
    z1 = z1.reshape(xx.shape)
    #plt.figure(1, figsize=(4, 3))
    plt.pcolormesh(xx1,yy1,z1,cmap=plt.cm.Paired)
    plt.scatter(Xtrain[:,0],Xtrain[:,1],c=ytrain.ravel(),edgecolors="k",cmap=plt.cm.Paired)
    plt.xlabel('Age')
    plt.ylabel('Estimated Salary')

    plt.xlim(xx1.min(), xx1.max())
    plt.ylim(yy1.min(), yy1.max())
    plt.xticks(())
    plt.yticks(())
    plt.show()
    
    y_pred1 = classifier1.predict(Xtest)
    print(metrics.confusion_matrix(ytest,y_pred1))
